## Usage

```js
var merge = require('{%= name %}');

var obj = {};
merge(obj, 'foo', {a: 'b'})
// obj.foo ==> {a: 'b'}

var obj = {foo: {a: 'b'}};
merge(obj, 'foo', {c: 'd'})
// obj.foo ==> {a: 'b', c: 'd'}

var obj = {};
merge(obj, 'a.b.c', {one: 'two'});
merge(obj, 'a.b.c', {three: 'four'});
// obj.a.b.c ==> {one: 'two', three: 'four'}

var obj = {foo: {bar: {baz: {a: 'b'}}}};
merge(obj, 'foo', {bar: {baz: {c: 'd'}}});
//=> {foo: {bar: {baz: {a: 'b', c: 'd'}}}}
```
